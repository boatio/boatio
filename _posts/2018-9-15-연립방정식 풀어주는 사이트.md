<script>
    function solve(){
        var data = document.getElementById('boxer').value;
        document.getElementById('whatisX').innerHTML = 'x = '+ nerdamer.solveEquations(data.split(",")).toString().replace("/","÷").replace("[","").replace("]","");
    }
</script>
